window.handlePunch = function(type) { const select = document.getElementById('employeeSelect'); const empName = select.value; if (!empName) return alert("⚠️ 請先在最上方選擇您的名字！"); if (!navigator.geolocation) return alert("您的瀏覽器不支援定位功能！"); navigator.geolocation.getCurrentPosition((position) => { const distance = getDistanceFromLatLonInKm(COMPANY_LAT, COMPANY_LON, position.coords.latitude, position.coords.longitude); // 只有「上班(in)」才檢查距離 if (type === 'in' && distance > MAX_DISTANCE_KM) { return alert(`打卡失敗！您目前距離公司超過範圍。`); } const dept = select.options[select.selectedIndex].getAttribute('data-dept'); const now = new Date(); const timeNum = now.getHours() * 100 + now.getMinutes(); let punchStatus = "正常"; let displayType = type === 'in' ? "上班" : "下班"; // 🎯 全新升級的打卡時間判斷邏輯 if (type === 'in') { // 上班打卡 if (timeNum < 720) { // ★ 文字已更新為「因個人因素提早上班」 const choice = prompt("目前早於 07:20，請選擇原因：\n1. 因個人因素提早上班"); if (choice === "1") { punchStatus = "個人因素提早上班"; // 寫入資料庫的狀態也同步更新 } else { return alert("未選擇原因，打卡取消。"); } } else if (timeNum > 730) { const choice = prompt("已超過 07:30，請選擇狀態：\n1. 遲到\n2. 正常上班"); if (choice === "1") punchStatus = "遲到"; else if (choice === "2") punchStatus = "正常上班"; else return alert("未選擇原因，打卡取消。"); } else { punchStatus = "正常上班"; } } else { // 下班打卡 if (timeNum >= 1650) { const choice = prompt("已超過 16:50 下班時間，請選擇原因：\n1. 經主管事前核准後同意加班\n2. 因個人因素延後下班"); if (choice === "1") punchStatus = "主管核准加班"; else if (choice === "2") punchStatus = "個人因素晚走"; else return alert("未選擇原因，打卡取消。"); } else { punchStatus = "正常下班"; } } // 發送資料到 Google 雲端 fetch(CONFIG_URL, { method: "POST", body: JSON.stringify({ action: "write", sheetName: "attendance", name: empName, dept: dept, type: displayType, status: punchStatus }) }).then(() => alert(`✅ ${empName} 雲端打卡成功！\n狀態：${punchStatus}`)) .catch(err => alert("網路連線錯誤，請稍後再試。")); }, () => alert("定位失敗，請允許網頁存取位置資訊！"), { enableHighAccuracy: true }); }
